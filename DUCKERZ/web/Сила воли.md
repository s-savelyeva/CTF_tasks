```
import requests
import json


def direct_server_request():
    url = 'http://tasks.duckerz.ru:30069/check_clicks'

    # Отправляем сразу 600000 кликов
    data = {'clicks': 600000}

    response = requests.post(url, json=data)

    if response.status_code == 200:
        result = response.json()
        print(f"Ответ сервера: {result}")
        return result
    else:
        print(f"Ошибка: {response.status_code}")
        return None


# Пробуем прямой запрос
```
