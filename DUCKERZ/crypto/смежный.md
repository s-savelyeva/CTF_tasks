Это аффинный шифр, скрипт для решения:

def affine_decrypt(ciphertext, a, b):
    """Дешифрует аффинный шифр"""

    # Находим мультипликативное обратное
    def mod_inverse(a, m):
        for i in range(1, m):
            if (a * i) % m == 1:
                return i
        return None

    a_inv = mod_inverse(a, 26)
    if a_inv is None:
        raise ValueError(f"Нет обратного для {a} по модулю 26")

    plaintext = ""

    for letter in ciphertext:
        if letter.isalpha():
            if letter.islower():
                enc = ord(letter) - ord('a')
                tmp = (a_inv * (enc - b)) % 26
                plaintext += chr(tmp + ord('a'))
            else:
                enc = ord(letter) - ord('A')
                tmp = (a_inv * (enc - b)) % 26
                plaintext += chr(tmp + ord('A'))
        else:
            plaintext += letter

    return plaintext


# Зашифрованный текст

# Дешифруем
plaintext = affine_decrypt(ciphertext, a, b)
print(f"Зашифрованный: {ciphertext}")
print(f"Расшифрованный: {plaintext}")
